cmake_minimum_required(VERSION 3.5)

project(MarkdownConverter)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt5 REQUIRED COMPONENTS WebEngineWidgets Core Gui Widgets WebEngine)

add_executable(MarkdownConverter main.cpp)

target_link_libraries(MarkdownConverter 
    Qt5::Core 
    Qt5::Gui 
    Qt5::Widgets 
    Qt5::WebEngineWidgets 
    Qt5::WebEngine
)
