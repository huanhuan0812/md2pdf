cmake_minimum_required(VERSION 3.16)

project(MarkdownConverter
    VERSION 1.0
    LANGUAGES CXX
)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找Qt6包
find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets WebEngineWidgets WebEngineCore WebEngineQuick)

# 添加可执行文件
qt_add_executable(MarkdownConverter
    main.cpp
)

# 设置目标属性
target_compile_definitions(MarkdownConverter PRIVATE
    QT_CORE_LIB
    QT_GUI_LIB
    QT_WIDGETS_LIB
    QT_WEBENGINEWIDGETS_LIB
    QT_WEBENGINECORE_LIB
)

# 链接Qt库
target_link_libraries(MarkdownConverter PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::WebEngineWidgets
    Qt6::WebEngineCore
)

# 安装规则
install(TARGETS MarkdownConverter
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    BUNDLE DESTINATION .
)

# Windows特定设置
if(WIN32)
    # 自动部署Qt运行时
    include(GNUInstallDirs)
    install(CODE "
        include(BundleUtilities)
        fixup_bundle(\"${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_BINDIR}/MarkdownConverter.exe\" 
                    \"\" 
                    \"${QT6_INSTALL_PREFIX}\")
    ")
endif()
